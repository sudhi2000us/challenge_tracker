// Challenger names and their progress (this can be dynamically set from a data source or form) const challengers = [ { name: "Vikaus", activity: "running", distance: 10, hours: 0, points: 0 }, { name: "Aryan", activity: "cycling", distance: 20, hours: 0, points: 0 }, { name: "Sudheesh", activity: "swimming", distance: 5, hours: 0, points: 0 }, { name: "Suhas", activity: "walking", distance: 15, hours: 0, points: 0 }, { name: "Sriram", activity: "squash", distance: 0, hours: 2, points: 0 }, { name: "Vasu", activity: "gym", distance: 0, hours: 1, points: 0 }, { name: "Rashmi", activity: "running", distance: 12, hours: 0, points: 0 }, ]; // Points calculation for each activity const activityPoints = { "walking": 0.5, "running": 1, "cycling": 0.25, "swimming": 8, "squash": 4, "gym": 5 }; // Success threshold const successThreshold = 50; // Function to calculate points function calculatePoints(challenger) { if (challenger.activity === "walking" || challenger.activity === "running" || challenger.activity === "cycling" || challenger.activity === "swimming") { return challenger.distance * activityPoints[challenger.activity]; } else if (challenger.activity === "squash" || challenger.activity === "gym") { return challenger.hours * activityPoints[challenger.activity]; } return 0; } // Function to check if the challenger has successfully earned 50 points function checkSuccess(points) { return points >= successThreshold ? "Success" : "Keep Going"; } // Dynamically set progress for each challenger const container = document.querySelector('.container'); challengers.forEach(challenger => { const points = calculatePoints(challenger); // Calculate points based on activity challenger.points = points; // Store points const status = checkSuccess(points); // Check if the challenger has succeeded const challengerDiv = document.createElement('div'); challengerDiv.classList.add('challenger'); challengerDiv.innerHTML = `

${challenger.name}

Activity: ${challenger.activity.charAt(0).toUpperCase() + challenger.activity.slice(1)}

Progress: ${points} Points - ${status}

`; container.appendChild(challengerDiv); });
